#ifndef ZWIERZE_H
#define ZWIERZE_H

using namespace std;

class Zwierze
{
    public:
        Zwierze(string imie);
        ~Zwierze();
        void ustawImie(string imie);
        string naLancuch() const;
        virtual string dajRodzajZwierza() const;

    protected:

    private:
        string mImie;
};

#endif // ZWIERZE_H

///////////

#include "Zwierze.h"
#include <iostream>
#include <string.h>

using namespace std;

Zwierze::Zwierze(string imie)
{
    this->mImie = imie;
}

Zwierze::~Zwierze()
{
    //dtor
}

void Zwierze::ustawImie(string imie)
{
    this->mImie = imie;
}

string Zwierze::naLancuch() const
{
    return dajRodzajZwierza() + mImie;
}

string Zwierze::dajRodzajZwierza() const
{
    return "Zwierz: ";
}
